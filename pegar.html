function toggleResumo() { const container = document.getElementById("resumojogos-charts-container"); const button = document.querySelector(".toggle-button"); if (container.style.maxHeight === "0px" || container.style.maxHeight === "") { container.style.maxHeight = "1000px"; /* Expande o container */ button.textContent = "Ocultar Resumo"; } else { container.style.maxHeight = "0px"; /* Oculta o container */ button.textContent = "Mostrar Resumo"; } }
